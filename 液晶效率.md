$$
\begin{align}
T\%_{all} &= Pol_{all} \times LC\% \times AR\% \times CF\% \\
T\%_1 &= Pol_{TFT} \times AR\% \times CF\% \\
T\%_2 &= AR\% \times CF\% \\
\frac{T\%_{all}}{T\%_2} &= Pol_{all} \times LC\%  \\
\frac{T\%_{all}}{T\%_1} &= Pol_{CF} \times LC\% 
\end{align}
$$
